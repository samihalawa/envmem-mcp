Quick Start

# Add to your MCP config (Claude Desktop, etc.)
{
  "mcpServers": {
    "envmem": {
      "url": "https://envmem.dev/mcp",
      "headers": {
        "x-api-key": "your-api-key-here"
      }
    }
  }
}

# Or use with Smithery
npx @anthropic/smithery install envmem

# Import your .env file
import_env_variables({
  envText: `
    OPENAI_API_KEY=sk-...
    STRIPE_SECRET_KEY=sk_live_...
    SUPABASE_URL=https://...
  `
})

# Then search naturally
search_env_variables({ query: "payment processing" })
// Returns: STRIPE_SECRET_KEY, STRIPE_WEBHOOK_SECRET...
